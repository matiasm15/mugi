def initialize(param, size = nil)
  case param
  when Integer
    @size = size || 8
    @n = param
  when Array
    @size = size || param.sum(&:size)
    @n = param.reverse.inject do |acu, i|
      i.to_word + acu
    end.to_i
  when Word
    @size = size || param.size
    @n = param.to_i
  end

  @n %= BASE**@size
end

def partition(value)
  new_size = @size / value
  last_index = value - 1

  (0..last_index).map do |i|
    Word.new(@n >> (new_size * (last_index - i)), new_size)
  end
end
