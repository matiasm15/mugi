def initialize(key, iv)
  key = key.bytes if key.is_a?(String)
  key = Word.new(key, UNIT_SIZE * 2)

  iv = iv.bytes if iv.is_a?(String)
  iv = Word.new(iv, UNIT_SIZE * 2)

  first_step(key)
  second_step(key, iv)
  third_step
end
